        \textbf{Оптимизация кода}
        \smallskip

        Оптимизация кода:

        \begin{enumerate}
            \item{На естественном языке отимизация не имеет строгих правил и 
            сводится к редактированию;}
            \item{Для программ она предполагает следующее:
            
            \begin{enumerate}
                \item{Увеличение скорости работы программы;}
                \item{Уменьшение объёма используемой памяти;}
                \item{И так далее.}
            \end{enumerate}
            }
        \end{enumerate}
