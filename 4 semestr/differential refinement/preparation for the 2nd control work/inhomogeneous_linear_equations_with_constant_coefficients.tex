\section{Неоднородные линейные уравнения с постоянными
коэффициентами}

    \subsection{Как выглядит неоднородное линейное уравнение с
    постоянными коэффициентами?}

        Однородное линейное уравнение с постоянными
        коэффициентами имеет вид:

        \[
            y^{(n)} + a_1 y^{(n - 1)} + \dots +
            a_{n - 1} y^{'} + a_n y = f(x),  
        \]
        
        где $a_1, \dots, a_n$~--- заданные числа.

    \subsubsection{Теория}    

        Подобные уравнения решаются по следующему
        алгоритму:

        \begin{enumerate}
            \item{Этап
            \begin{enumerate}
                \item{Выписываем соответствующее однородное
                уравнение.

                Оно имеет вид

                \[
                    y^{(n)} + a_1 y^{(n - 1)} + \dots +
                    a_{n - 1} y^{'} + a_n y = 0  
                \]
                }
                \item{Решаем его по уже описанным ранее
                алгоритмам.

                В итоге получим его общее, оно имеет вид:

                \[
                    y = c_1 y_1 + c_2 y_2 + \dots + c_{k}
                    y_{k}.
                \]
                }
            \end{enumerate}
            }
            \item{Этап
            \begin{enumerate}
                \item{Найдём частное решение исходного
                неоднородного уравнения. Для этого
                будем считать, что
                $c_1, \dots, c_k$, как пока что неизвестные
                функции.}
                \item{
                    \[
                        y_h = c_1(x) y_1 + c_2(x) y_2 + \dots
                        + c_{k}(x) y_{k}.  
                    \]
                }
                \item{Составляем систему уравнений от 
                $c_1^{'} (x), \dots c_k^{'} (x)$ и находим их.

                Система будет иметь вид:

                \begin{equation*}
                    \begin{cases}
                        c_1^{'}(x) y_1 + \dots + c_k^{'} y_k = 0\\
                        c_1^{'}(x) y_1^{'} + \dots + c_k^{'} y_k^{'} = 0\\
                        \dots\\
                        c_1^{'}(x) y_1^{(k-1)} + \dots + c_k^{'} y_k^{(k-1)} = f(x)\\
                    \end{cases}
                \end{equation*}
                
                }
            \end{enumerate}
            }
            \item{Этап
            
            В качестве ответа записывает следующее:
            
            Ответ: $y = c_1 y_1 + c_2 y_2 + \dots + c_{k}
            y_{k} + y_h.$
            }
        \end{enumerate}

    \subsection{Практика}
    
        \textbf{Пример 1:}
        \bigskip

        Дано:

        \[
            y^{''} - 2 y^{''} - 3 y = e^{4x}  
        \]

        Решение:

        \begin{enumerate}
            \item{Этап
            \begin{enumerate}
                \item{Выписываем соответствующее однородное
                уравнение

                \[
                    y^{''} - 2 y^{''} - 3 y = 0
                \]
                }
                \item{Решаем соответствующее однородное
                уравнение

                \[
                    \lambda^2 - 2 \lambda - 3 = 0  
                \]
                \[
                    D = 4 - 4 (-3) = 4^2  
                \]
                \[
                    \lambda_1 = \frac{2 + 4}{2} = 3, \;\;\;
                    \lambda_2 = \frac{2 - 4}{2} = -1  
                \]

                Общее решение однородного уравнения:

                \[
                    y = c_1 e^{3x} + c_2 e^{-x}  
                \]
                }
            \end{enumerate}
            }
            \item{Этап
            \begin{enumerate}
                \item{Выписываем частное решение
                    \[
                        y_h = c_1(x) e^{3x} + c_2(x)e^{-x},  
                    \]

                    где $c_1(x), c_2(x)$~--- пока ещё неизвестные
                    функции.
                }
                \item{Составляем систему уравнений относительно
                $c_1^{'}(x)$ и $c_2^{'}(x)$ и находим их.

                Процесс нахождения $c_1^{'}(x)$ и $c_2^{'}(x)$ опустим,
                так как я всё"=таки не бессмертен)

                \begin{equation*}
                    \begin{cases}
                        c_1^{'}(x) e^{3x} + c_2^{'}(x) e^{-x} = 0\\
                        c_1^{'}(x) 3 e^{3x} - c_2^{'}(x) e^{-x} = e^{4x}\\
                    \end{cases}
                \end{equation*}

                
                \[
                    \dots
                \]

                \begin{equation*}
                    \begin{cases}
                        c_1^{'}(x) = \frac{e^x}{4}\\
                        c_2^{'}(x) = - \frac{e^{5x}}{4}\\
                    \end{cases}
                \end{equation*}
                }
                \item{Находим $c_1(x), c_2(x)$
                
                \[
                    c_1 (x) = \frac{1}{4} \int e^{x} dx = 
                    \frac{e^{x}}{4}
                \]
                \[
                    c_2(x) = -\frac{1}{4} \int e^{5x} dx = 
                    -\frac{e^5x}{20} 
                \]
                }
                \item{Записываем полученное частное решение
                    
                \[
                    y_h = \frac{e^{x}}{4} - \frac{e^5x}{20}
                \]
                }
            \end{enumerate}
            }
            \item{Этап
            
                Ответ: $y = c_1 e^{3x} + c_2 e^{-x} +
                \frac{e^{x}}{4} - \frac{e^5x}{20}$
            }
        \end{enumerate}

        \textbf{Пример 2:}
        \bigskip

        Дано:

        \[
            y^{''} + y = 4 \sin{x}  
        \]

        Решение:

        \begin{enumerate}
            \item{Этап
            \begin{enumerate}
                \item{Выписываем соответствующее однородное
                уравнение

                \[
                    y^{''} + y = 0
                \]
                }
                \item{Решаем соответствующее однородное
                уравнение

                \[
                    \lambda^2 + 1 = 0  
                \]
                \[
                    \lambda^2 = -1
                \]
                \[
                    \lambda_1 = i, \;\;\;
                    \lambda_2 = -i  
                \]

                ФСР:

                \[
                    y_1 = e^{ix}, \;\;\; y_2 = e^{-ix}  
                \]

                Преобразуем комплексные корни к 
                вещественным.

                Рассмотрим $y_1$, а $y_2$ отбрасываем, как
                комплексно сопряжённый с $y_1$:

                \[
                    y_1 = e^{ix} = e^0 (\cos{x} + \sin{x}) =
                    \cos{x} + \sin{x}  
                \]

                Общее решение однородного уравнения:

                \[
                    y = c_1 \cos{x} + c_2 \sin{x}  
                \]
                }
            \end{enumerate}
            }
            \item{Этап
            \begin{enumerate}
                \item{Выписываем частное решение
                    \[
                        y_h = c_1(x) \cos{x} + c_2(x) \sin{x},  
                    \]

                    где $c_1(x), c_2(x)$~--- пока ещё неизвестные
                    функции.
                }
                \item{Составляем систему уравнений относительно
                $c_1^{'}(x)$ и $c_2^{'}(x)$ и находим их.

                Процесс нахождения $c_1^{'}(x)$ и $c_2^{'}(x)$ так же
                опускаем.

                \begin{equation*}
                    \begin{cases}
                        c_1^{'}(x) \cos{x} + c_2^{'}(x) \sin{x} = 0\\
                        - c_1^{'}(x) \sin{x} + c_2^{'}(x) \cos{x} = 4 \sin{x}\\
                    \end{cases}
                \end{equation*}

                
                \[
                    \dots
                \]

                \begin{equation*}
                    \begin{cases}
                        c_1^{'}(x) = -4 \sin^2{x}\\
                        c_2^{'}(x) = 4 \sin{x} \cos{x}\\
                    \end{cases}
                \end{equation*}
                }
                \item{Находим $c_1(x), c_2(x)$
                
                \[
                    c_1 (x) = -4 \int \sin^2{x} dx = 
                    \sin{2x} - 2x
                \]
                \[
                    c_2(x) = 4 \int \sin{x} \cos{x} dx = 
                    2 \sin^2{x} 
                \]
                }
                \item{Записываем полученное частное решение
                    
                \[
                    y_h = \sin{2x} \cos{x} - 2x \cos{x} + 2 \sin^3{x}
                \]
                }
            \end{enumerate}
            }
            \item{Этап
            
                Ответ: $y = c_1 \cos{x} + c_2 \sin{x} +
                \sin{2x} \cos{x} - 2x \cos{x} + 2 \sin^3{x}$
            }
        \end{enumerate}
