\section{Линейные дифференциальные уравнения n"=го порядка}

    \subsection{Определения}

        \subsubsection{Линейные дифференциальные уравнения n"=го порядка
        (однородные и неоднородные)}

            \textbf{Определение линейного однородного дифференциального уравнения
            n"=го порядка}
                \smallskip

                Однородное дифференциальное уравнение n"=го порядка имеет вид:

                \[
                    a_0(x)y^{(n)}(x) + a_1(x)y^{(n - 1)}(x) + \dots + 
                    a_n(x)y = 0,  
                \]

                где $a_0(x), \dots, a_n(x)$~--- заданные непрерывные
                функции.
                \bigskip

            \textbf{Определение линейного неоднородного дифференциального уравнения
                n"=го порядка}   
                \smallskip

                Неоднородное дифференциальное уравнение n"=го порядка имеет вид:

                \[
                    a_0(x)y^{(n)}(x) + a_1(x)y^{(n - 1)}(x) + \dots + 
                    a_n(x)y = f(x),  
                \]

                где $a_0(x), \dots, a_n(x), f(x)$~--- заданные непрерывные
                функции.

        \subsubsection{Задача Коши для дифференциального уравнения n"=го порядка}
        
            \[
                (1) \;\;\; y^{(n) + a_1(x)y^{(n - 1)} + \dots +
                a_n(x)y = f(x)}  
            \]
            \bigskip

            \textbf{Определение задачи Коши для дифференциального уравнения n"=го
            порядка}
                \smallskip

                Задача Коши для линейного уравнения (1) имеет вид:

                \begin{equation*}
                    \begin{cases}
                        y^{(n)} + a_1(x)y^{(n - 1)}(x) + \dots + a_n(x)y = f(x)\\
                        y(x_0) = y_0\\
                        y^{'}(x_0) = y^{'}_0\\
                        \dots\\
                        y^{(n-1)}(x_0) = y^{(n - 1)}_0\\
                    \end{cases}
                \end{equation*}

                где $y(x_0) = y_0, \dots, y^{(n-1)}(x_0) = y^{(n - 1)}_0$~---
                начальные условия и $x_0, y_0, \dots, y^{(n - 1)}_0$~--- заданные
                числа.

        \subsubsection{Линейно зависимые и линейно независимые функции}
        
            \textbf{Определение линейно зависимых функций}
                \smallskip

                Функции $\varphi_1(x), \varphi_2(x), \dots, \varphi_m(x)$ называются
                линейно зависимыми на отрезке $[a, b]$, если найдутся константы
                $\alpha_1, \alpha_2, \dots, \alpha_m$~--- числа, не все равные нулю,
                такие, что линейная комбинация:

                \[
                    \alpha_1\varphi(x) + \dots + \alpha_m\varphi(m) \equiv 0  
                \]

            \textbf{Определение линейно независимых функций}
                \smallskip
                
                Функции $\varphi_1(x), \dots, \varphi_m(x)$ называются 
                линейно независимыми на отрезке $[a, b]$, если 

                \[
                    \alpha_1\varphi_1(x) + \dots + \alpha_m\varphi_m(x) \equiv 0
                    \Leftrightarrow \alpha_1 = \dots = \alpha_m = 0  
                \]

        \subsubsection{Определитель Вронского функций $\varphi_1(x),\varphi_2(x),
        \dots, \varphi_m(x)$}
            
            \textbf{Определение определителя Вронского функций
            $\varphi_1(x),\varphi_2(x), \dots, \varphi_m(x)$}
                \smallskip

                Определителем Вронского функций $\varphi_1(x),\varphi_2(x),
                \dots, \varphi_m(x)$ называется:

                \[
                \begin{vmatrix}
                    \varphi_1(x)\ \varphi_2(x)\ \dots\ \varphi_m(x)\\
                    \varphi^{'}_1(x)\ \varphi^{'}_2(x)\ \dots\ \varphi^{'}_m(x)\\
                    \dots \\
                    \varphi^{(n - 1)}_1(x)\ \varphi^{(n - 1)}_2(x)\ \dots\ \varphi^{(n - 1)}_m(x)\\
                \end{vmatrix}
                \]

        \subsubsection{Фундаментальная система решения}
        
            \textbf{Опредеение фундаментальной системы решений однородного
            линейного уравнения}
                \smallskip

                Фундаментальной системой решений однородного линейного уравнения
                $l(y) = 0$ называется:

                $\varphi_1(x), \dots, \varphi_n(x)$~--- линейно независимые и их
                количество совпадает с порядком уравнения.
                
        \subsubsection{Линейные дифференциальные уравнения с постоянными
        коэффициентами}
            
            \textbf{Определение однородного линейного уравнения с постоянными
            коэффициентами}
                \smallskip

                Линейное однородное уравнение с постоянными коэффициентами
                имеет вид:

                \[
                    a_0y^{(n)} + a_1 y^{(n - 1)} + \dots + a_n y = 0,  
                \]

                где $a_0, \dots, a_n$~--- заданные числа.
                \bigskip

            \textbf{Определение неоднородного линейного уравнения с 
            постоянными коэффициентами}
                \smallskip
                
                Линейное неоднородноре уравнение с постоянными коэффициентами
                имеет вид:

                \[
                    a_0y^{(n)} + a_1 y^{(n - 1)} + \dots + a_n y = f(x),  
                \]

                где $a_0, \dots, a_n$~--- заданные числа.
                \bigskip

        \subsubsection{Необходимое условие линейной зависимости}
        
            \textbf{Теорема (необходимое условие линейной зависимости)}
                \smallskip

                Если функции $\varphi_1(x), \dots, \varphi_n(x)$~---
                линейно зависимы на $[a, b]$. Тогда их определитель
                вронского тождественно равен нулю.

                \[
                    \mathbf{W} \equiv 0  
                \]

        \subsubsection{Условие линейной независимости решений}
        
            Рассмотрим линейное уравнение:

            \[
                (1) \;\;\; y^{(n)} + a_1(x) y^{(n - 1)} + \dots +
                a_m(x)y = 0; \;\;\; a \leq x \leq b
            \]
            \bigskip

            \textbf{Теорема (условие линейной независимости решений линейного 
            уравнения)}
                \smallskip

                Пусть $\varphi_1(x), \dots, \varphi_n(x)$~--- линейно независимые
                решения уравнения (1). Тогда

                \[
                    \forall x \in [a, b] \;\;\; \mathbf{W} \neq 0
                \]

                Доказательство (нужно или нет?:))

        \subsubsection{Свойство линейности и следствия из него}

            \[
                y^{(n)} + a_1(x) y^{(n - 1)} + \dots + a_m(x)y = f(x)
            \]

            Обозначим $^{(n)} + a_1(x) y^{(n - 1)} + \dots + a_m(x)y$ как
            $l(y)$.
            \bigskip

            \textbf{Свойство линейности $l(y)$}
                \smallskip

                \[
                    l(y_1(x) + y_2(x)) = l(y_1(x)) + l(y_2(x));  
                \]

                \[
                    l(\sum^m_{k = 1} \alpha_k \varphi_k(x)) = \sum^m_{k = 1}
                    \alpha_k l(\varphi_k(x))
                \]
                \bigskip

            \textbf{Следствие свойства линейности}
                \smallskip
                
                Пусть $\varphi_1(x), \dots, \varphi_n(x)$~--- решения уравнения
                $l(y) = 0$. Тогда $\varphi^{\circ}(x) = \alpha_1\varphi_1(x) +
                \dots + \alpha_m\varphi_m(x)$ тоже является решением этого
                уравнения.        
            
                Доказательство (нужно или нет?:))


        \subsubsection{Существование ФСР}
            
            \textbf{Теорема (о существовании ФСР у любого однородного
            линейного уравнения)}
                \smallskip

                Фундаментальная система решений существует для любого
                однородного линейного уравнения.

        \subsubsection{Вид общего решения однородноого линейного уравнения}        
                
            Рассмотрим однородное линейное уравнение

            \[
                (1) \;\;\; y^{(n)} + a_1(x) y^{(n - 1)} + \dots + a_m(x)y = 0;
                \;\;\; a \leq x \leq b
            \]

            или 

            \[
                l(y) = 0
            \]

            \textbf{Теорема (вид общего решения однородного линейного уравнения)}
                \smallskip

                Пусть $\varphi_1(x), \dots, \varphi_n(x)$~--- ФСР. Тогда
                общее решение уравнения (1) имеет вид:

                \[
                    y = c_1\varphi_1(x) + \dots + c_n\varphi_n(x),  
                \]

                где $c_1, \dots, c_n$~--- произвольные константы.

        \subsubsection{Вид общего решения неоднородноого линейного уравнения} 
        
            Рассмотрим неоднородное линейное уравнение

                \[
                    (1) \;\;\; y^{(n)} + a_1(x) y^{(n - 1)} + \dots + a_m(x)y = f(x);
                    \;\;\; a \leq x \leq b
                \]

                или 

                \[
                    l(y) = f(x)
                \]

                и соответствующее ему однородное линейное уравнение 

                \[
                    (2) \;\;\; l(y) = 0  
                \]

                \textbf{Теорема (вид общего решения неоднородного линейного уравнения)}
                    \smallskip

                    Пусть $\varphi_1(x), \dots, \varphi_n(x)$~--- ФСР (2) и
                    $y_h(x)$~--- частное решение (1). Тогда
                    общее решение уравнения (1) имеет вид:

                    \[
                        y = c_1\varphi_1(x) + \dots + c_n\varphi_n(x) + y_h(x),  
                    \]

                    где $c_1, \dots, c_n$~--- произвольные константы.

                    \textbf{В $y_h(x)$ вместо h должна стоять буква ч, но латех этого
                    сделать не позволяет.}
