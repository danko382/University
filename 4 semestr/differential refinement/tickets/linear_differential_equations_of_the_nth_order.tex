\section{Линейные дифференциальные уравнения n"=го порядка}

    \subsection{Определения}

        \subsubsection{Линейные дифференциальные уравнения n"=го порядка
        (однородные и неоднородные)}

            \textbf{Определение линейного однородного дифференциального уравнения
            n"=го порядка}
                \smallskip

                Однородное дифференциальное уравнение n"=го порядка имеет вид:

                \[
                    a_0(x)y^{(n)}(x) + a_1(x)y^{(n - 1)}(x) + \dots + 
                    a_n(x)y = 0,  
                \]

                где $a_0(x), \dots, a_n(x)$~--- заданные непрерывные
                функции.
                \bigskip

            \textbf{Определение линейного неоднородного дифференциального уравнения
                n"=го порядка}   
                \smallskip

                Неоднородное дифференциальное уравнение n"=го порядка имеет вид:

                \[
                    a_0(x)y^{(n)}(x) + a_1(x)y^{(n - 1)}(x) + \dots + 
                    a_n(x)y = f(x),  
                \]

                где $a_0(x), \dots, a_n(x), f(x)$~--- заданные непрерывные
                функции.

        \subsubsection{Задача Коши для дифференциального уравнения n"=го порядка}
        
            \[
                (1) \;\;\; y^{(n) + a_1(x)y^{(n - 1)} + \dots +
                a_n(x)y = f(x)}  
            \]
            \bigskip

            \textbf{Определение задачи Коши для дифференциального уравнения n"=го
            порядка}
                \smallskip

                Задача Коши для линейного уравнения (1) имеет вид:

                \begin{equation*}
                    \begin{cases}
                        y^{(n)} + a_1(x)y^{(n - 1)}(x) + \dots + a_n(x)y = f(x)\\
                        y(x_0) = y_0\\
                        y^{'}(x_0) = y^{'}_0\\
                        \dots\\
                        y^{(n-1)}(x_0) = y^{(n - 1)}_0\\
                    \end{cases}
                \end{equation*}

                где $y(x_0) = y_0, \dots, y^{(n-1)}(x_0) = y^{(n - 1)}_0$~---
                начальные условия и $x_0, y_0, \dots, y^{(n - 1)}_0$~--- заданные
                числа.

        \subsubsection{Линейно зависимые и линейно независимые функции}
        
            \textbf{Определение линейно зависимых функций}
                \smallskip

                Функции $\varphi_1(x), \varphi_2(x), \dots, \varphi_m(x)$ называются
                линейно зависимыми на отрезке $[a, b]$, если найдутся константы
                $\alpha_1, \alpha_2, \dots, \alpha_m$~--- числа, не все равные нулю,
                такие, что линейная комбинация:

                \[
                    \alpha_1\varphi(x) + \dots + \alpha_m\varphi(m) \equiv 0  
                \]

            \textbf{Определение линейно независимых функций}
                \smallskip
                
                Функции $\varphi_1(x), \dots, \varphi_m(x)$ называются 
                линейно независимыми на отрезке $[a, b]$, если 

                \[
                    \alpha_1\varphi_1(x) + \dots + \alpha_m\varphi_m(x) \equiv 0
                    \Leftrightarrow \alpha_1 = \dots = \alpha_m = 0  
                \]

        \subsubsection{Определитель Вронского функций $\varphi_1(x),\varphi_2(x),
        \dots, \varphi_m(x)$}
            
            \textbf{Определение определителя Вронского функций
            $\varphi_1(x),\varphi_2(x), \dots, \varphi_m(x)$}
                \smallskip

                Определителем Вронского функций $\varphi_1(x),\varphi_2(x),
                \dots, \varphi_m(x)$ называется:

                \[
                \begin{vmatrix}
                    \varphi_1(x)\ \varphi_2(x)\ \dots\ \varphi_m(x)\\
                    \varphi^{'}_1(x)\ \varphi^{'}_2(x)\ \dots\ \varphi^{'}_m(x)\\
                    \dots \\
                    \varphi^{(n - 1)}_1(x)\ \varphi^{(n - 1)}_2(x)\ \dots\ \varphi^{(n - 1)}_m(x)\\
                \end{vmatrix}
                \]

        \subsubsection{Фундаментальная система решения}
        
            \textbf{Опредеение фундаментальной системы решений однородного
            линейного уравнения}
                \smallskip

                Фундаментальной системой решений однородного линейного уравнения
                $l(y) = 0$ называется:

                $\varphi_1(x), \dots, \varphi_n(x)$~--- линейно независимые и их
                количество совпадает с порядком уравнения.
                
        \subsubsection{Линейные дифференциальные уравнения с постоянными
        коэффициентами}
            
            \textbf{Определение однородного линейного уравнения с постоянными
            коэффициентами}
                \smallskip

                Линейное однородное уравнение с постоянными коэффициентами
                имеет вид:

                \[
                    a_0y^{(n)} + a_1 y^{(n - 1)} + \dots + a_n y = 0,  
                \]

                где $a_0, \dots, a_n$~--- заданные числа.
                \bigskip

            \textbf{Определение неоднородного линейного уравнения с 
            постоянными коэффициентами}
                \smallskip
                
                Линейное неоднородноре уравнение с постоянными коэффициентами
                имеет вид:

                \[
                    a_0y^{(n)} + a_1 y^{(n - 1)} + \dots + a_n y = f(x),  
                \]

                где $a_0, \dots, a_n$~--- заданные числа.
                \bigskip

        \subsubsection{Необходимое условие линейной зависимости}
        
            \textbf{Теорема (необходимое условие линейной зависимости)}
                \smallskip

                Если функции $\varphi_1(x), \dots, \varphi_n(x)$~---
                линейно зависимы на $[a, b]$. Тогда их определитель
                вронского тождественно равен нулю.

                \[
                    \mathbf{W} \equiv 0  
                \]

        \subsubsection{Условие линейной независимости решений}
        
            Рассмотрим линейное уравнение:

            \[
                (1) \;\;\; y^{(n)} + a_1(x) y^{(n - 1)} + \dots +
                a_m(x)y = 0; \;\;\; a \leq x \leq b
            \]
            \bigskip

            \textbf{Теорема (условие линейной независимости решений линейного 
            уравнения)}
                \smallskip

                Пусть $\varphi_1(x), \dots, \varphi_n(x)$~--- линейно независимые
                решения уравнения (1). Тогда

                \[
                    \forall x \in [a, b] \;\;\; \mathbf{W} \neq 0
                \]

                Доказательство (нужно или нет?:))

        \subsubsection{Свойство линейности и следствия из него}

            \[
                y^{(n)} + a_1(x) y^{(n - 1)} + \dots + a_m(x)y = f(x)
            \]

            Обозначим $^{(n)} + a_1(x) y^{(n - 1)} + \dots + a_m(x)y$ как
            $l(y)$.
            \bigskip

            \textbf{Свойство линейности $l(y)$}
                \smallskip

                \[
                    l(y_1(x) + y_2(x)) = l(y_1(x)) + l(y_2(x));  
                \]

                \[
                    l(\sum^m_{k = 1} \alpha_k \varphi_k(x)) = \sum^m_{k = 1}
                    \alpha_k l(\varphi_k(x))
                \]
                \bigskip

            \textbf{Следствие свойства линейности}
                \smallskip
                
                Пусть $\varphi_1(x), \dots, \varphi_n(x)$~--- решения уравнения
                $l(y) = 0$. Тогда $\varphi^{\circ}(x) = \alpha_1\varphi_1(x) +
                \dots + \alpha_m\varphi_m(x)$ тоже является решением этого
                уравнения.        
            
                Доказательство (нужно или нет?:))


        \subsubsection{Существование ФСР}
            
            \textbf{Теорема (о существовании ФСР у любого однородного
            линейного уравнения)}
                \smallskip

                Фундаментальная система решений существует для любого
                однородного линейного уравнения.

        \subsubsection{Вид общего решения однородноого линейного уравнения}        
                
            Рассмотрим однородное линейное уравнение

            \[
                (1) \;\;\; y^{(n)} + a_1(x) y^{(n - 1)} + \dots + a_m(x)y = 0;
                \;\;\; a \leq x \leq b
            \]

            или 

            \[
                l(y) = 0
            \]

            \textbf{Теорема (вид общего решения однородного линейного уравнения)}
                \smallskip

                Пусть $\varphi_1(x), \dots, \varphi_n(x)$~--- ФСР. Тогда
                общее решение уравнения (1) имеет вид:

                \[
                    y = c_1\varphi_1(x) + \dots + c_n\varphi_n(x),  
                \]

                где $c_1, \dots, c_n$~--- произвольные константы.

        \subsubsection{Вид общего решения неоднородноого линейного уравнения} 
        
            Рассмотрим неоднородное линейное уравнение

            \[
                (1) \;\;\; y^{(n)} + a_1(x) y^{(n - 1)} + \dots + a_m(x)y = f(x);
                \;\;\; a \leq x \leq b
            \]

            или 

            \[
                l(y) = f(x)
            \]

            и соответствующее ему однородное линейное уравнение 

            \[
                (2) \;\;\; l(y) = 0  
            \]

            \textbf{Теорема (вид общего решения неоднородного линейного уравнения)}
                \smallskip

                Пусть $\varphi_1(x), \dots, \varphi_n(x)$~--- ФСР (2) и
                $y_h(x)$~--- частное решение (1). Тогда
                общее решение уравнения (1) имеет вид:

                \[
                    y = c_1\varphi_1(x) + \dots + c_n\varphi_n(x) + y_h(x),  
                \]

                где $c_1, \dots, c_n$~--- произвольные константы.

                \textbf{В $y_h(x)$ вместо h должна стоять буква ч, но латех этого
                сделать не позволяет.}

        \subsection{Алгоритм метод вариации нахождения частного решения}
        
            \textbf{Метод вариации произвольных постоянных нахождения $y_h(x)$}
                \smallskip

                По теореме о виде общего решения линейного однородного уравнения
                $y_0 = c_1\varphi_1(x) + \dots + c_n\varphi_n(x)$~--- общее
                решение уравнения $l(y) = 0$.

                Будем искать частное решение $y_h(x)$ в виде:

                \[
                    y_h(x) = c_1(x)\varphi_1(x) + \dots + c_n(x)\varphi_n(x),  
                \]

                где $c_1(x), \dots, c_n(x)$~--- пока неизвестные функции.
                \bigskip

                Рассмотрим систему уравнений

                \begin{equation*}
                    (5)
                    \begin{cases}
                        c_1^{'}(x)\varphi_1(x) + \dots + c_n^{'}(x)\varphi_n(x) = 0\\
                        c_1^{'}(x)\varphi_1^{'}(x) + \dots + c_n^{'}(x)\varphi_n^{'}(x) = 0\\
                        c_1^{'}(x)\varphi_1^{''}(x) + \dots + c_n^{'}(x)\varphi_n^{''}(x) = 0\\
                        \dots\\
                        c_1^{'}(x)\varphi_1^{(n - 2)}(x) + \dots + c_n^{'}(x)\varphi_n^{(n - 2)}(x) = 0\\
                        c_1^{'}(x)\varphi_1^{(n - 1)}(x) + \dots + c_n^{'}(x)\varphi_n^{(n - 1)}(x) = f(x)\\
                    \end{cases}
                \end{equation*}

                Пусть $a \leq x \leq b$, x~--- фиксированное $\Rightarrow$
                (5)~--- линейная алгебраическая система относительно
                $c^{'}_i(x), \;\;\; i = \overline{1, n}$

                Определитель системы $= \mathbf{W}(x) \Rightarrow$
                $\mathbf{W}(x) \neq 0 \Rightarrow$ по теореме из алгебры (5)
                имеет единственное решение.

                По формулам Крамера:

                \[
                    c^{'}_1(x) = \frac{
                        \begin{vmatrix}
                            0 \ \dots \\
                            \dots \\
                            f \ \dots \\
                        \end{vmatrix}
                    }
                        {\mathbf{W}} = \frac{\mathbf{W}_1(x)}{\mathbf{W}(x)}
                \]
                \[
                        \dots
                \]
                \[
                    (6) \;\;\; c^{'}_n(x) = \frac{
                        \begin{vmatrix}
                            \dots \ 0 \\
                            \dots \\
                            \dots \ f \\
                        \end{vmatrix}
                    }
                        {\mathbf{W}} = \frac{\mathbf{W}_n(x)}{\mathbf{W}(x)}
                \]

                Таким образом, $c^{'}_i(x) \;\;\; i = \overline{1, n}$ находится
                по формуле (6) $\Rightarrow$ 

                \[
                        \Rightarrow \;\;\; c_k(x) = \int \frac{\mathbf{W}_k(x)}
                        {\mathbf{W}(x)} dx
                \]

        \subsection{Обоснование метода вариации}
        
            Рассмотрим частный случай $n = 2$:

            \[
                (1) \;\;\; y^{''} + a_1(x)y^{'} + a_2(x)y = f(x)
            \]
            \[
                (2) \;\;\; y^{''} + a_1(x)y^{'} + a_2(x)y = 0
            \]
            \[
                \varphi_1(x), \varphi_2(x)
            \]~--- ФСР (2)

            \[
                y_0 = c_1 \varphi_1(x) + c_2 \varphi_2(x)
            \]
            \[
                y_h = c_1(x)\varphi_1(x) + c_2(x)\varphi_2(x)
            \]

            \begin{equation*}
                (5)
                \begin{cases}
                    c_1^{'}(x)\varphi_1(x) + c_2^{'}(x)\varphi_2(x) = 0\\
                    c_1^{'}(x)\varphi_1^{'}(x) + c_2^{'}(x)\varphi^{'}_2(x) = f(x)\\
                \end{cases}
            \end{equation*}

            \[
                c_1(x) = \int \frac{\mathbf{W_1}(x)}{\mathbf{W}(x)}dx \;\;\;
                c_2(x) = \int \frac{\mathbf{W_2}(x)}{\mathbf{W}(x)}dx  
            \]

            Подставляем $c_1(x)$ и $c_2(x)$ в (4) $\Rightarrow$

            Покажем, что полученное $y_h$ является решением (1):

            \[
                y^{'}_h = c_1^{'}(x)\varphi_1(x) + c_1(x)\varphi_1^{'}(x) +
                c_2^{'}(x)\varphi_2(x) + c_2(x)\varphi_2^{'}(x) = c_1(x)\varphi_1^{'}(x)
                +  c_2(x)\varphi^{'}_2(x)  
            \]
            \[
                y_h^{''} = c^{'}_1(x)\varphi^{'}_1(x) + c_1\varphi^{''}_1(x)
                + c^{'}_2(x)\varphi^{'}_2(x) + c_2(x)\varphi^{''}_2(x) =
                f(x) + c_1(x)\varphi^{''}_1(x) + c_2(x)\varphi^{''}_2(x) = y_h^{''} 
                \Rightarrow 
            \]
            \[
                f(x) + c_1(x)\varphi^{''}_1(x) + c_2(x)\varphi^{''}_2(x) +
                a_1(x)(c_1(x)\varphi^{'}_1(x) + c_2(x)\varphi^{'}_2(x)) +
                a_2(x)(c_1(x)\varphi_1(x) + c_2(x)\varphi_2(x)) =
            \]
            \[
                = f(x) + c_1(x)(\varphi^{''}_1(x) + a_1(x)\varphi_1^{'}(x) +
                a_2(x)\varphi_1(x)) + a_2(x)(\varphi^{''}_2(x) + a_1(x)\varphi^{''}_1(x) +
                a_2(x)\varphi_2(x)) = f(x)  
            \]

            То есть

            \[
                l(y_h) \equiv f(x)  
            \]

        \subsection{Формула Остроградского"=Луивилля}
            
            Пусть $y_1(x), \dots, y_n(x)$~--- решение однородного линейного
            уравнения: $y^{(n)} + a_1(x)y^{(n - 1)} + \dots +
            a_n(x)y = 0 \;\;\; a \leq x \leq b$.

            Рассмотрим определитель Вронского:

            \[
                \mathbf{W}(x) = 
                \begin{vmatrix}
                    y_1(x) \ \dots \ y_n(x)\\
                    \ \dots\ \\
                    y_1^{(n - 1)}(x) \ \dots \ y_n^{(n - 1)}(x)\\
                \end{vmatrix}
                =
                \sum (\pm) y^{(k1)}_{j_1}(x)y^{(k2)}_{j_2}(x) \dots y^{(kn)}_{j_n}(x)
            \]

            \textbf{Сама формула}

            Пусть $x_0$ произвольная точка из $[a, b]$. Тогда:

            \[
                \mathbf{W}(x) = \mathbf{W}(x_0)e^{-\int^x_{x_0}a_1(t)dt}
            \]

        \subsection{Метод Эйлера (случай простых корней)}
    
            \[
                (1) \;\;\; a_0y^{(n)} + a_1y^{(n - 1)} + \dots + a_ny = 0,
            \]
        
            где $a_0, \dots, a_n$~--- заданные числа.
            \bigskip

            \textbf{Метод Эйлера решения однородного линейного уравнения с 
            постоянными коэффициентами (случай простых корней)}
            \smallskip

                \begin{enumerate}
                    \item{Ищем частное решение уравнения (1) в виде
                    \[
                        y(x) = e^{\lambda x},    
                    \]

                    $\lambda$~--- число.
                    }
                    \item{Вычисляем формулы
                    
                    \[
                        y^{'}(x) = \lambda e^{\lambda x}  
                    \]
                    \[
                        y^{''}(x) = \lambda^2 e^{\lambda x}  
                    \]
                    \[
                        \dots  
                    \]
                    \[
                        y^{(n)}(x) = \lambda^n e^{\lambda x}  
                    \]
                    }
                    \item{Подставляем эти формулы в (1):
                    
                    \[
                        a_0\lambda^ne^{\lambda x} + a_1\lambda^{n - 1}e^{\lambda x} +
                        \dots + a_ne^{\lambda x} \equiv 0    
                    \]
                    \[
                        e^{\lambda x}(a_0 \lambda^n + \dots + a_n) \equiv 0 
                    \]
                    }
                    \item{так как $e^{\lambda x} \neq 0$, то
                    
                    \[
                        y = e^{\lambda x}~---  
                    \]

                    является решением уравнения (1) $\Leftrightarrow \;\;\; \lambda$
                    является корнем алгебраического уравнения:

                    \[
                        (2) \;\;\; a_0 \lambda^n + \dots + a_n = 0  
                    \]
                    }
                    \item{(2) имеет n простых корней $\lambda_1, \dots, 
                    \lambda_n \Rightarrow$ функции $y_1 = e^{\lambda_1 x}, 
                    y_2 = e^{\lambda_2 x}, \dots, y_n = e^{\lambda_n x}$~---
                    решение (1).
                    }
                    \item{Поажем, что ${e^{\lambda_k x}}^n_{k = 1}$~--- ФСР (1)
                    
                    Рассмотрим определитель Вронского $\mathbf{W}$

                    \[
                        \mathbf{W}(x) =
                        \begin{vmatrix}
                            e^{\lambda_1 x} \ \dots \ e^{\lambda_n x}\\
                            \lambda_1 e^{\lambda_1 x} \ \dots \ \lambda_n e^{\lambda_n x}\\
                            \ \dots \ \\
                            \lambda_1^{(n - 1)} e^{\lambda_1 x} \ \dots \ \lambda_n^{(n - 1)} e^{\lambda_n x}\\
                        \end{vmatrix}  
                        =
                        e^{\lambda_1 x} e^{\lambda_2 x} \dots e^{\lambda_n x} 
                        \begin{vmatrix}
                            1 \ \dots \ 1\\
                            \lambda_1 \ \dots \ \lambda_n\\
                            \ \dots \ \\
                            \lambda_1^{(n - 1)}\ \dots \ \lambda_n^{(n - 1)}\\
                        \end{vmatrix} =
                    \]
                    \[
                        = e^{\lambda_1 x} e^{\lambda_2 x} \dots e^{\lambda_n x} \prod 
                        (\lambda_j - \lambda_k) \neq 0 \Rightarrow 
                    \]
                    ${e^{\lambda_k x}}^n_{k = 1}$~--- ФСР уравнения (1) $\Rightarrow$
                    общее решение (1) имеет вид:

                    \[
                        y = c_1e^{\lambda_1 x} + \dots + c_ne^{\lambda_n x}
                    \]
                    }
                \end{enumerate}
