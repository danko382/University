\section{Линейные системы}

    \subsection{Определения}

        \subsubsection{Нормальная линейная система, её векторная запись}

            \textbf{Нормальная неоднородная линейная система}
            \smallskip

                Нормальная неоднородная линейная система имеет вид:

                \begin{equation*}
                    (1)
                    \begin{cases}
                        y_1^{'} = a_{11}(x)y_1 + a_{12}(x)y_2 + \dots + a_{1n}(x)y_n + f_1(x);\\
                        y_2^{'} = a_{21}(x)y_1 + a_{22}(x)y_2 + \dots + a_{2n}(x)y_n + f_2(x);\\
                        \dots\\
                        y_n^{'} = a_{n1}(x)y_1 + a_{n2}(x)y_2 + \dots + a_{nn}(x)y_n + f_n(x);\\
                    \end{cases},
                \end{equation*}

                где $x$~--- неизветсные переменные $a \leq x \leq b$,
                $a_{ij}(x), f_{i}(x)$~--- заданные непрерывные функции,
                $y_{i}$~--- неизвестные функции.

            \textbf{Векторная запись нормальной неоднородной линейной системы}
            
                Обозначим

                $$
                    Y(x) = 
                    \begin{pmatrix}
                        y_1(x)\\
                        \dots\\
                        y_n(x)
                    \end{pmatrix}  
                $$~--- вектор"=функция

                $$
                F(x) =
                \begin{pmatrix}
                    f_1(x)\\
                    \dots\\
                    f_n(x)\\
                \end{pmatrix}
                $$~--- вектор"=функция

                $$
                A(x) = 
                \begin{pmatrix}
                    a_{11}(x)\ \dots \ a_{1n}(x)\\
                    \dots\\
                    a_{n1}(x)\ \dots \ a_{nn}(x)\\
                \end{pmatrix}
                $$~--- матрица"=функция

                По определению:

                $$
                Y^{'}(x) =
                \begin{pmatrix}
                    y^{'}_1(x)\\
                    \dots\\
                    y^{'}_n(x)
                \end{pmatrix}
                $$

                $$
                A(x)Y(x) = 
                \begin{pmatrix}
                    a_{11}(x)y_1 + \dots + a_{1n}(x)y_n\\
                    \dots\\
                    a_{n1}(x)y_1 + \dots + a_{nn}(x)y_n\\
                \end{pmatrix}
                $$

                $A(x)Y(x) + F(x)$~--- вектор соответствующий правой части (1).
                Тогда 

                $$
                    Y^{'} = A(x)Y + F(X)
                $$~--- векторная запись (1).
                \bigskip

            \textbf{Определение нормальной однородной линейной системы}
                \smallskip        

                Линейная система (1) называется однородной, если $\forall f_n(x) = 0
                \;\;\; i = \overline{1, n}$

                \[
                    (2) \;\;\; Y^{'} = A(x)Y  
                \]

                или 

                \begin{equation*}
                    \begin{cases}
                        y_1^{'} = a_{11}(x)y_1 + a_{12}(x)y_2 + \dots + a_{1n}(x)y_n;\\
                        y_2^{'} = a_{21}(x)y_1 + a_{22}(x)y_2 + \dots + a_{2n}(x)y_n;\\
                        \dots\\
                        y_n^{'} = a_{n1}(x)y_1 + a_{n2}(x)y_2 + \dots + a_{nn}(x)y_n;\\
                    \end{cases},
                \end{equation*}

                где $x$~--- неизветсные переменные $a \leq x \leq b$,
                $a_{ij}(x)$~--- заданные непрерывные функции,
                $y_{i}$~--- неизвестные функции.

        \subsection{Решение системы}
        
            \textbf{Определение решения нормальной линейной системы}
            \smallskip

            Говорят, что дифференцируемые функции $\varphi_1(x), \dots, 
            \varphi_n(x)$ образуют решение системы (1), если при подстановке их
            в уравнение вместо соответствующих $y$ получается тождество:

            \[
                y^{'}_k = a_{k1}(x)\varphi_1(x) + dots + a_{kn}\varphi_n(x) +
                f_k(x); \;\;\; k = \overline{1, n}
            \]

            В векторной форме:

            Дифференцируемая функция $\Phi(x) = 
            \begin{pmatrix}
                \varphi_1(x)\\
                \dots\\
                \varphi_n(x)\\
            \end{pmatrix}
            $ является решением (1), если: 

            \[
                \Phi^{'}(x) = A(x)\Phi(x) + F(x)
            \]

        \subsection{Задача Коши для системы}

            \begin{equation*}
                (1)
                \begin{cases}
                    y_1^{'} = a_{11}(x)y_1 + a_{12}(x)y_2 + \dots + a_{1n}(x)y_n + f_1(x);\\
                    y_2^{'} = a_{21}(x)y_1 + a_{22}(x)y_2 + \dots + a_{2n}(x)y_n + f_2(x);\\
                    \dots\\
                    y_n^{'} = a_{n1}(x)y_1 + a_{n2}(x)y_2 + \dots + a_{nn}(x)y_n + f_n(x);\\
                \end{cases},
            \end{equation*}
        
            \textbf{Определение задачи Коши нормальной линейной системы}
            \smallskip

                Пусть $x_0 \in [a, b]$, $y^0_1, \dots, y^0_n$~--- заданные числа.

                Начальными условиями для системы (1) в точке $x_0$ называются
                равенства:

                \[
                    y_1(x_0) = y^0_1  
                \]
                \[
                    y_2(x_0) = y^0_2 
                \]
                \[
                    \dots  
                \]
                \[
                    y_n(x_0) = y^0_n 
                \]

                Задача 1"=3 называется задачей Коши для системы (1).

                В векторной форме:

                \[
                    Y^{'} = A(x)Y(x) + F(x), \;\;\; Y(x_0) = Y^0,  
                \]

                где $Y^0 = 
                \begin{pmatrix}
                    y^0_1\\
                    \dots\\
                    y^0_n\\
                \end{pmatrix}$

        \subsection{Линейно зависимые и линейно независимые вектор"=функции}
        
            \textbf{Определение линейно зависимых вектор"=функций}
            \smallskip
            
                Пусть $\Phi_1(x), \dots, \Phi_n(x) \in R^n, \;\;\; a \leq x \leq b$.
                Эти вектор"=функции называются линейно зависимыми на $[a, b]$,
                если:

                $\exists \alpha_1, \dots, \alpha_n$~--- числа не все равные 0 :
                $\alpha_1\Phi_1(x) + \alpha_n\Phi_n(x) = 0$
                \bigskip

            \textbf{Определение линейно независимых вектор"=функций}
            \smallskip
            
                Пусть $\Phi_1(x), \dots, \Phi_n(x) \in R^n, \;\;\; a \leq x \leq b$.
                Эти вектор"=функции называются линейно независимыми на $[a, b]$,
                если:

                $\alpha_1\Phi_1(x) + \alpha_n\Phi_n(x) = 0 \;\;\;
                \Leftrightarrow \;\;\; \forall \alpha_i = 0 \;\;\; i = \overline{1, n}$

        \subsection{Определитель Вронского вектор"=функций}
        
            \textbf{Определение определителя Вронского вектор"=функций}
            \smallskip

                Пусть $\Phi_1(x), \dots, \Phi_n(x) \in R^n$ их определителем
                Вронского называется следующий определитель:

                \[
                    W = |\Phi_1(x) \Phi_2(x) \dots \Phi_n(x)|  
                \]

        \subsection{Фундаментальная система решений линейной системы}
        
            \textbf{Определение фундаментальной системы решений нормальной
            линейной системы}
            \smallskip

            Рассмотрим (1) $Y^{'} = A(x)Y(x), \;\;\; a \leq x \leq b, \;\;\;
            Y(x) \in R^n$ $A(x)$~--- размер $nxn$.

            Линейно независимые решения $\Phi_1(x), \dots, \Phi_n(x)$ называются
            фундаментальной системой решений.


        \subsection{Фундаментальная матрица решений однородной системы}
        
            \textbf{Определение фундаментальной матрицы решений нормальной
            однородной системы}

                Матрица $T(x) = (Y_1(x)Y_2(x) \dots Y_n(x))$ называется 
                фундаментальной матрицей для однородной системы $Y^{'} =
                A(X)Y(x), \;\;\; a \leq x \leq b$, если:

                $$Y_1(x), \dots, Y_n(x)$$~--- ФСР однородной системы.

        \subsection{Линейные системы с постоянными коэффициентами}
        
            \textbf{Определение однородной линейной системы с постоянными 
            коэффициентами}
            \smallskip

                Линейная однородная система с постоянными коэффициентами имеет вид:

                \begin{equation*}
                    \begin{cases}
                        y_1^{'} = a_{11}(x)y_1 + a_{12}(x)y_2 + \dots + a_{1n}(x)y_n;\\
                        y_2^{'} = a_{21}(x)y_1 + a_{22}(x)y_2 + \dots + a_{2n}(x)y_n;\\
                        \dots\\
                        y_n^{'} = a_{n1}(x)y_1 + a_{n2}(x)y_2 + \dots + a_{nn}(x)y_n;\\
                    \end{cases},
                \end{equation*}

                $a_{ij}$~--- заданные числа.

                В векторной записи:

                $$Y^{'} = AY$$~--- где $A = (a_{ij})^n_{ij = 1}, Y =
                \begin{pmatrix}
                    y_1(x)\\
                    \dots\\
                    y_n(x)\\
                \end{pmatrix}
                $
