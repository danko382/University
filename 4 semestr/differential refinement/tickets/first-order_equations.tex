\section{Уравнения первого порядка}

    \subsection{Определения}

        \subsubsection{Обыкновенные дифференциальные уравнения
        первого порядка}

            \textbf{Определение обыкновенного дифференциального уравнения
            первого порядка}
                \smallskip

                Обыкновенным дифференциальным уравнением первого
                порядка называется уравнение вида

                \[
                    F(x, y, y^{'}) \equiv 0  
                \]

        \subsubsection{Частное решение обыкновенного дифференциального уравнения
        первого порядка}
        
            \textbf{Определение частного решения обыкновенного дифференциального
            уравнения первого порядка}
                \smallskip

                $(1) \;\;\; F(x, y, y^{'}) \equiv 0$~--- обыкновенное
                дифференциальное уравнение первого порядка.
                \bigskip

                Частным решением обыкновенного дифференциального
                уравнения первого порядка называется непрерывно 
                дифференцируемая функция $\varphi(x)$, при подстановки
                которой в уравнение (1) получим тождество

                \[
                    \varphi^{'}(x) \equiv f(x, \varphi(x))
                \] 

        \subsubsection{Общее решение обыкновенного дифференциального уравнения
        первого порядка}

            \textbf{Определение общего решения обыкновенного дифференциального
                уравнения первого порядка}
                    \smallskip

                    Множество всех решений обыкновенного дифференциального
                    уравнения первого порядка называется его общим решением.

        \subsubsection{Общий интеграл дифференциального уравнения}

        \subsubsection{Уравнение с разделяющимися переменными первого 
        порядка}

            \textbf{Определение уравнения сразделяющимися переменными
            первого порядка}
                \smallskip

                Дифференциальным уравннием с разделяющимися переменными
                первого порядка называется уравнение вида

                \[
                    y^{'} = f_1(x) f_2(y),  
                \]

                где $f_1(x)$ и $f_2(y)$~--- заданные функции.

        \subsubsection{Обыкновенное дифференциальное уравнение первого порядка
        в симметричной форме}
        
            \textbf{Определение обыкновенного дифференциального уравнения
            первого порядка в симметричной форме}
                \smallskip

                Обыкновенное дифференциальное уравнение первого порядка
                в симметрицной форме имеет вид 

                \[
                    A(x, y)dx + B(x, y)dy = 0,  
                \]

                где $A$ и $B$~--- заданные функции двух переменных,
                причём переменные $x$ и $y$ равноправны.

        \subsubsection{Линейное дифференциальное уравнение первого порядка}

            \textbf{Определение линейного дифференциального уравнения
            первого порядка}
                \smallskip

                Линейным дифференциальным уравнением первого порядка
                называется уравнение вида

                \[
                    a_0(x)Y^{'} + a_1(x)y = f(x),  
                \]

                где $x$~--- неизвестнвя пееменная, $y = y(x)$~--- неизывестная 
                функция, $a_0(x)$ и $a_1(x)$~--- известные непрерывные функции.
        
        
        \subsubsection{Задача Коши для дифференциального уравнения 
        первого порядка}

            \textbf{Определение задачи Коши для дифференциального уравнения 
            первого порядка}
                \smallskip

                Пусть $y^{'} = f(x, y)$~--- дифференциальное
                уравнение первого порядка и $y(x_0) = y_0$~--- его начальное
                условие. Тогда задача Коши для него имеет вид

                
                \begin{equation*}
                    \begin{cases}
                        y^{'} = f(x, y)\\
                        y(x_0) = y_0
                    \end{cases}
                \end{equation*}

                где $x_0, y_0$~--- заданные числа.
