\section{Уравнения первого порядка}

    \subsection{Определения}

        \subsubsection{Обыкновенные дифференциальные уравнения
        первого порядка}

            \textbf{Определение обыкновенного дифференциального уравнения
            первого порядка}
                \smallskip

                Обыкновенным дифференциальным уравнением первого
                порядка называется уравнение вида
                \[
                    F(x, y, y^{'}) \equiv 0  
                \]

        \subsubsection{Частное решение обыкновенного дифференциального уравнения
        первого порядка}
        
            \textbf{Определение частного решения обыкновенного дифференциального
            уравнения первого порядка}
                \smallskip

                \begin{equation}
                    F(x, y, y^{'}) \equiv 0
                    \label{eqn:common_diff_eqn}
                \end{equation}
                обыкновенное дифференциальное уравнение первого порядка.
                \bigskip

                Частным решением обыкновенного дифференциального
                уравнения первого порядка называется непрерывно 
                дифференцируемая функция $\varphi(x)$, при подстановке
                которой в уравнение (\ref{eqn:common_diff_eqn}) получим тождество
                \[
                    \varphi^{'}(x) \equiv f(x, \varphi(x))
                \] 

        \subsubsection{Общее решение обыкновенного дифференциального уравнения
        первого порядка}

            \textbf{Определение общего решения обыкновенного дифференциального
                уравнения первого порядка}
                    \smallskip

                    Множество всех решений обыкновенного дифференциального
                    уравнения первого порядка называется его общим решением.

        \subsubsection{Общий интеграл дифференциального уравнения первого порядка}

            Дифференциальное уравнение первого порядка, разрешенное относительно производной,
            имеет вид
            \begin{equation}
                y^{'} = f(x,y)
                \label{eqn:resolved_derivative}
            \end{equation}
            \bigskip

            \textbf{Определение общего интеграла дифференциального уравнения первого
            порядка:}
                \smallskip

                Общее решение уравнения (\ref{eqn:resolved_derivative}) в неявном виде
                \[
                    F(x, y) = C  
                \]
                называется общим интегралом уравнения (\ref{eqn:resolved_derivative}).

        \subsubsection{Уравнение с разделяющимися переменными первого 
        порядка}

            \textbf{Определение уравнения с разделяющимися переменными
            первого порядка}
                \smallskip

                Дифференциальным уравннием с разделяющимися переменными
                первого порядка называется уравнение вида
                \[
                    y^{'} = f_1(x) f_2(y),  
                \]
                где $f_1(x)$ и $f_2(y)$~--- заданные функции.

        \subsubsection{Обыкновенное дифференциальное уравнение первого порядка
        в симметричной форме}
        
            \textbf{Определение обыкновенного дифференциального уравнения
            первого порядка в симметричной форме}
                \smallskip

                Обыкновенное дифференциальное уравнение первого порядка
                в симметрицной форме имеет вид 
                \[
                    A(x, y)dx + B(x, y)dy = 0,  
                \]
                где $A$ и $B$~--- заданные функции двух переменных,
                причём переменные $x$ и $y$ равноправны.

        \subsubsection{Линейное дифференциальное уравнение первого порядка}

            \textbf{Определение линейного дифференциального уравнения
            первого порядка}
                \smallskip

                Линейным дифференциальным уравнением первого порядка
                называется уравнение вида
                \[
                    a_0(x)y^{'} + a_1(x)y = f(x),  
                \]
                где $x$~--- неизвестнвя переменная, $y = y(x)$~--- неизвестная 
                функция, $a_0(x)$ и $a_1(x)$~--- известные непрерывные функции.
        
        
        \subsubsection{Задача Коши для дифференциального уравнения 
        первого порядка}

            \textbf{Определение задачи Коши для дифференциального уравнения 
            первого порядка}
                \smallskip

                Пусть $y^{'} = f(x, y)$~--- дифференциальное
                уравнение первого порядка и $y(x_0) = y_0$~--- некоторое начальное
                условие. Тогда задача Коши для него имеет вид
                $$
                    \begin{cases}
                        y^{'} = f(x, y)\\
                        y(x_0) = y_0
                    \end{cases}
                $$
                где $x_0, y_0$~--- заданные числа.

    \subsection{Теоремы и алгоритмы}
        
        \subsubsection{Алгоритм решения уравнений с
        разделяющимися переменными первого порядка}

            \begin{enumerate}
                \item Переходим к дифференциалам:
                \[
                    y^{'} = \frac{dy}{dx} \;\;\;\;\;\; \frac{dy}{dx} = f_1(x) f_2(y);
                \]

                \item Делим переменные:
                \[
                    \frac{dy}{dx} = f_1(x) f_2(y) \;\;\; \left| \cdot dx \frac{1}{f_2(y)} \right.
                \]

                \[
                    \frac{dy}{f_2(y)} = f_1(x)dx    
                \]

                \item Вычисляем интегралы:
                \[
                    \int \frac{dy}{f_2(y)} = \int f_1(x) dx + C,  
                \]
                где $C \in R$ и $\int \frac{dy}{f_2(y)} = F_2(y)$,
                $\int f_1(x) dx = F_1(x)$.
                
                Получим:
                \begin{equation}
                    F_2(y) = F_1(x) + C
                \end{equation}

                \item Разрешаем последнее уравнение относительно $y$:
                \begin{equation}
                    y = \varphi(x, C), \;\;\; C \in R,
                \end{equation}
                где $\varphi(x, C)$~--- общее решение.

                \item Ответ: $\varphi(x, C)$.
            \end{enumerate}
            
        \subsubsection{Метод вариации решния линейного дифференциального
        уравнения первого порядка.}    

            Линейное дифференциальное уравнение первого порядка имеет вид:
            \[
                a_0(x)y^{'} + a_1(x)y = f(x),  
            \]

            Тогда
            \[
                y^{'} + \frac{a_1(x)}{a_0(x)}y = \frac{f(x)}{a_0(x)}  
            \]

            Обозначим $p(x) = \frac{a_1(x)}{a_0(x)}, \;\;\; q(x) = \frac{f(x)}{a_0(x)}$.

            Следовательно уравнение примет вид:
            \begin{equation}
                y^{'} + p(x)y = q(x), \;\;\; a \leq x \leq b  
                \label{eqn:first_order_linear_inhomogeneous}
            \end{equation}
            
            \textbf{Метод вариации произвольной постоянной:}
                \smallskip

                \begin{enumerate}
                    \item Этап:
                    
                    Найдём общее решение соответствующего (\ref{eqn:first_order_linear_inhomogeneous})
                    однородного уравнения:
                    
                    \begin{equation}
                        y^{'} + p(x)y = 0
                        \label{eqn:first_order_linear_homogeneous}
                    \end{equation}
                    \[
                        y^{'} = - p(x)y  
                    \]
                    \[
                        \frac{dx}{dy} = - p(x)y  
                    \]
                    \[
                        \int \frac{dy}{y} = - \int p(x)dx + C  
                    \]
                    \[
                        \ln{|y|} = -F_1(x) + C  
                    \]
                    \[
                        y = \pm e^{C} e^{-F_1(x)}  
                    \]

                    Получим: $y_\text{о} = C e^{-F_1(x)}$~--- общее решение (\ref{eqn:first_order_linear_homogeneous}).
                    \item Этап: 
                    
                    Ищем частное решение (\ref{eqn:first_order_linear_inhomogeneous}) в виде:
                    \begin{equation}
                        y_\text{ч} = C(x)e^{-F_1(x)},
                        \label{eqn:particular_solution}
                    \end{equation}
                    где $C(x)$~--- пока неизвестная функция.
                    \[
                        y^{'} = C^{'}(x)e^{-F_1(x)} + C(x) e^{-F_1(x)}(-p(x))  
                    \]

                    Подставляем в (\ref{eqn:first_order_linear_inhomogeneous}):
                    \[
                        C^{'(x)}e^{-F_1(x)} - C(x)e^{-F_1(x)}p(x) + p(x)C(x)
                        e^{-F_1(x)} = q(x)  
                    \]
                    \[
                        C^{'}(x) = e^{F_1(x)}q(x)    
                    \]
                    \[
                        C(x) = \int e^{F_1(x)}q(x)dx \pm c = F_2(x) + C  
                    \]

                    Подставим в (\ref{eqn:particular_solution}):
                    \[
                        y = e^{-F_1(x)}(F_2(x) + C)
                    \]

                    Таким образом, $y = e^{-F_1(x)}(F_2(x) + C)$~--- общее решение
                    уравнения (\ref{eqn:particular_solution}).
                \end{enumerate}

            
        \subsubsection{Основная теорема существования и единственности}

            \textbf{Теорема о существовании и единственности решения 
            задачи Коши:}
                \smallskip

                Пусть $f(x, y)$~--- непрерывная функция и у неё
                существует частная производная $f^{'}_y(x, y)$. Тогда
                $\forall (x_0, y_0)$ задача Коши имеет единственное решение.

        \subsubsection{Сведение задачи Коши к интегральному уравнению}
        
            Пусть $\varphi(x)$~--- решение задачи Коши:

            \begin{equation}
                \varphi^{'}(x) \equiv f(x, \varphi(x))
                \label{eqn:int_eqn_1}
            \end{equation}
            \begin{equation}
                \varphi(x_0) = y_0
                \label{eqn:int_eqn_2}
            \end{equation}

            Перепишем (\ref{eqn:int_eqn_1}) в виде:

            \begin{equation}
                \varphi^{'}(t) = f(t, \varphi(t))
                \label{eqn:int_eqn_3}
            \end{equation}

            Фиксируем произвольный $x$ и берём интеграл от (\ref{eqn:int_eqn_3}):
            \[
                \varphi^{'} = f(t, \varphi(t)) \;\;\; \left| \int^x_{x_0} \dots dt \right.
            \]
            \[
                \varphi(x) - \varphi(x_0) = \int^x_{x_0} f(t, \varphi(t)) dt
            \]
            \begin{equation}
                \varphi(x) = y_0 + \int^x_{x_0} f(t, \varphi(t)) dt
                \label{eqn:int_eqn_4}
            \end{equation}

            По определению (\ref{eqn:int_eqn_4}) означает, что $\varphi(x)$ является решением
            интегрального уравнения

            \begin{equation}
                y(x) = y_0 + \int^x_{x_0} f(t, \varphi(t)) dt
                \label{eqn:int_eqn_5}
            \end{equation}

            Все остальные рассуждения обратимы $\implies$ задача Коши $\sim$
            уравнению (\ref{eqn:int_eqn_5}).
